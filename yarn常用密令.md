# yarn常用命令

```apl
yarn install //创建node_modules
yarn init -y  //创建package.json
yarn add react-router-dom@5.2.0	      // 安装指定版本依赖包
npm install react-router-dom@5.2.0		// 安装指定版本依赖包
yarn remove react-router-dom	   // 删除依赖包
yarn upgrade react-router-dom    // 升级依赖包
yarn add koa2 -S                 // 安装到生产环境
yarn add koa2 -D                 // 安装到开发环境
```

# npm常用命令

```apl
npm install   // 创建node_modules
npm init -y		// 创建package.json
npm install -g yarn  // 安装yarn
npm install react-router-dom@5.2.0	  // 安装指定版本依赖包
npm install react-router-dom@5.2.0		// 安装指定版本依赖包
npm uninstall react-router-dom	   // 删除依赖包
npm updata react-router-dom        // 升级依赖包 
npm install koa2 --save/-S         // 安装到生产环境
npm install koa2 --save-dev/-D     // 安装到开发环境

```
